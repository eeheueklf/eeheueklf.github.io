мһҗл°”мҠӨнҒ¬лҰҪнҠё кё°мҙҲ

н”„лЎ нҠём—”л“ң мӨ‘мҡ” к°ңл…җ - Javascript лӮҙмҡ© м •лҰ¬ 1нҺёмһ…лӢҲлӢӨ. мӣ№ мӮ¬мқҙнҠёмқҳ ліөмһЎн•ң лҸҷмһ‘л“Өмқ„ кө¬нҳ„н•  мҲҳ мһҲлҠ” JavaScriptмқҳ кё°ліё к°ңл…җл¶Җн„° ліҖмҲҳ, мқҙлІӨнҠё, к°қмІҙк№Ңм§Җ лӢӨлӨ„лҙӨмҠөлӢҲлӢӨ.

JavaScriptлҠ” л¬ҙм—Үмқёк°Җ?вҖӢ

мһҗл°”мҠӨнҒ¬лҰҪнҠё(JavaScript, JS)лҠ” мӣ№нҺҳмқҙм§ҖлҘј кө¬м„ұн•ҳлҠ” м–ём–ҙлЎң, мӮ¬мҡ©мһҗмҷҖ мғҒнҳёмһ‘мҡ©н•  мҲҳ мһҲкІҢ н•ҳлҠ” н”„лЎңк·ёлһҳл°Қ м–ём–ҙмһ…лӢҲлӢӨ. ('мһҗл°”'мҠӨнҒ¬лҰҪнҠёмқҙм§Җл§Ң, мһҗл°”мҷҖлҠ” кҙҖл Ёмқҙ м—ҶмҠөлӢҲлӢӨ..)
HTMLл§Ңмқҙ м Ғмҡ©лҗң м •м Ғмқё мӣ№нҺҳмқҙм§ҖлҠ” лӮҙмҡ©кіј кө¬мЎ°л§Ң ліҙм—¬мӨ„ мҲҳ мһҲм§Җл§Ң, JSлҘј мӮ¬мҡ©н•ңлӢӨл©ҙ лӢӨм–‘н•ң кё°лҠҘмқ„ кө¬нҳ„н•  мҲҳ мһҲмҠөлӢҲлӢӨ.

  • лІ„нҠј нҒҙлҰӯ вҶ’ лӮҙмҡ©ліҖкІҪ
  • нҸј мһ…л Ҙ вҶ’ кІҖмҰқ
  • лҸҷм Ғ мҪҳн…җмё  мғқм„ұ
  • л№„лҸҷкё° нҶөмӢ 
  • лёҢлқјмҡ°м Җ API нҷңмҡ©
  • м• лӢҲл©”мқҙм…ҳ & нҡЁкіј
  • кІҪкі м°Ҫ, нҷ•мқём°Ҫ, мһ…л Ҙм°Ҫ
  • 2D/3D к·ёлһҳн”Ҫ

мҰү HTMLмқҙ лјҲ, CSSк°Җ мӮҙ, JSлҠ” мӣҖм§Ғмһ„мқ„ лӢҙлӢ№н•©лӢҲлӢӨ. лҝҗл§Ңм•„лӢҲлқј JS мң„м—җм„ң лҸҷмһ‘н•ҳлҠ” л§ҺмқҖ лқјмқҙлёҢлҹ¬лҰ¬мҷҖ н”„л Ҳмһ„мӣҢнҒ¬лҘј нҷңмҡ©н•ҳл©ҙ ліөмһЎн•ң кё°лҠҘлҸ„ нҺёлҰ¬н•ҳкІҢ кө¬нҳ„н•  мҲҳ мһҲмҠөлӢҲлӢӨ.

ліҖмҲҳ (Variables)вҖӢ

ліҖмҲҳлҠ” лҚ°мқҙн„°лҘј м ҖмһҘн•ҳлҠ” мғҒмһҗмһ…лӢҲлӢӨ. JSм—җм„ңлҠ” var, let, const нӮӨмӣҢл“ңлҘј мӮ¬мҡ©н•ҳм—¬ ліҖмҲҳлҘј м„ м–ён•©лӢҲлӢӨ. нғҖ м–ём–ҙм—җм„ңлҠ” int, char л“ұмңјлЎң ліҖмҲҳмқҳ нғҖмһ…мқ„ лҜёлҰ¬ м§Җм •н•ҳм§Җл§Ң, JSлҠ” ліҖмҲҳмқҳ нғҖмһ…мқҙ мң лҸҷм ҒмқҙлқјлҠ” нҠ№м§•мқ„ к°–кі  мһҲмҠөлӢҲлӢӨ.

var data1;
let data2;
const data3 = 1;

  • var
    м „м—ӯ лІ”мң„ (лӢЁ, н•ЁмҲҳ лӮҙм—җм„ң м„ м–ёлҗҳл©ҙ н•ЁмҲҳ лІ”мң„)
    м—…лҚ°мқҙнҠё/мһ¬м„ м–ё к°ҖлҠҘ
    нҳёмқҙмҠӨнҢ…(м„ м–ёмқҙ л§Ё мң„лЎң, мҙҲкё°к°’мқҖ undefined)

  • let
    м„ м–ёлҗң лё”лЎқ лӮҙм—җм„ңл§Ң мӮ¬мҡ© к°ҖлҠҘ
    м—…лҚ°мқҙнҠё к°ҖлҠҘ, мһ¬м„ м–ё л¶Ҳк°ҖлҠҘ
    нҳёмқҙмҠӨнҢ…лҗҳм§Җл§Ң мҙҲкё°нҷ” м•ҲлҗЁ

  • const мғҒмҲҳ к°’ мң м§Җ
    м„ м–ёлҗң лё”лЎқ лӮҙм—җм„ңл§Ң мӮ¬мҡ© к°ҖлҠҘ
    м—…лҚ°мқҙнҠё/мһ¬м„ м–ё л¶Ҳк°Җ (лӢЁ, к°қмІҙ мҶҚм„ұмқҖ ліҖкІҪ к°ҖлҠҘ)
    м„ м–ёмӢң л°ҳл“ңмӢң мҙҲкё°нҷ” н•„мҡ”


рҹҢө

JSм—җм„ң varмқҙ letмңјлЎң лҢҖмІҙлҗҳлҠ” м¶”м„ёмқҙкё° л•Ңл¬ём—җ, varмқ„ мӮ¬мҡ©н•  мқјмқҖ кұ°мқҳ м—ҶмҠөлӢҲлӢӨ.
л”°лқјм„ң мғҒмҲҳ, к°қмІҙ/л°°м—ҙ => const, к°’мқҙ л°”лҖҢлҠ” ліҖмҲҳ => letл§Ң кё°м–өн•ҳл©ҙ лҗ©лӢҲлӢӨ.
мҪ”л“ң м§Өл•ҢлҠ” constлҘј кё°ліёмңјлЎң м“°кі , ліҖмҲҳ к°’мқҙ л°”лҖ”л•Ңл§Ң letмңјлЎң л°”кҫёл©ҙ мўӢмҠөлӢҲлӢӨ.


ліҖмҲҳм—җ лӢҙмқ„ мҲҳ мһҲлҠ” к°’мқҳ мў…лҘҳ

  • String (л¬ёмһҗм—ҙ) : Hello
  • Number (мҲ«мһҗ) : 10 141.325
  • Boolean (м°ё/кұ°м§“) : true false
  • Array (л°°м—ҙ) : [1, 2, 3] ["кі м–‘мқҙ", "к°•м•„м§Җ", "л…ёлЈЁ"]
  • Object (к°қмІҙ) : {name: "кі м–‘мқҙ", age: 4}

мқҙлІӨнҠё (Events)вҖӢ

JSлҘј нҶөн•ҙ мӮ¬мҡ©мһҗмқҳ н–үлҸҷм—җ л”°лҘё лҸҷмһ‘мқ„ м„Өм •н•  мҲҳ мһҲмҠөлӢҲлӢӨ.
мҳҲмӢң) нҒҙлҰӯмӢң л©”мӢңм§Җ лқ„мҡ°кё°

document.querySelector("html").onclick = function () {
  alert("нҒҙлҰӯн–ҲмҠөлӢҲлӢӨ");
};

мЈјмҡ” мқҙлІӨнҠё мў…лҘҳ:

  • click: нҒҙлҰӯ мӢң
  • mouseover: л§Ҳмҡ°мҠӨлҘј мҳ¬л ёмқ„ л•Ң
  • keydown / keyup: нӮӨліҙл“ң мһ…л Ҙ
  • submit: нҸј м ңм¶ң
  • scroll: мҠӨнҒ¬лЎӨ
  • load: нҺҳмқҙм§Җ лЎңл“ң мҷ„лЈҢ
  • input: мһ…л Ҙк°’ ліҖкІҪ

к°қмІҙвҖӢ

к°қмІҙлҠ” кҙҖл Ёлҗң лҚ°мқҙн„°мҷҖ н•ЁмҲҳмқҳ м§‘н•©мһ…лӢҲлӢӨ. мһҗл°”мҠӨнҒ¬лҰҪнҠём—җм„ңлҠ” лӘЁл“  кІғмқҙ к°қмІҙлқјкі  н•  л§ҢнҒј мһҗмЈј м“°мқҙлҠ” к°ңл…җмһ…лӢҲлӢӨ.

const cat = {
  name: "кі м–‘мқҙ",
  age: 1,
  sayAge: function () {
    console.log(this.name + ' лӮҳмқҙлҠ” ' + this.age + 'мӮҙ мқҙлӢӨ');
  },
};

cat.sayAge(); // кі м–‘мқҙ лӮҳмқҙлҠ” 1мӮҙмқҙлӢӨ.
  • object - к°қмІҙ (cat)
  • property вҖ“ к°қмІҙ м•Ҳмқҳ лҚ°мқҙн„° (name, age)
  • method вҖ“ к°қмІҙ м•Ҳмқҳ н•ЁмҲҳ (sayAge)

лӢӨмқҢкіј к°ҷмқҙ к°қмІҙ л°°м—ҙлЎң мӮ¬мҡ©н•  мҲҳ мһҲмҠөлӢҲлӢӨ.

const animals = [
  { name: "л…ёлЈЁ", age: 4 },
  { name: "кі м–‘мқҙ", age: 1 },
  { name: "л§ҲлЈЁ", age: 5 },

animals.forEach((animal) => {
  console.log(`${animal.name}лҠ” ${animal.age}мӮҙмқҙлӢӨ.`);
});
